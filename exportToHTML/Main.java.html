import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        GestorPacientes gestor = new GestorPacientes();
        gestor.getPacientes().addAll(FicheiroPacientes.carregarPacientes());


        System.out.println("Quantos pacientes deseja registar?");
        int n = scanner.nextInt();
        scanner.nextLine();

        for (int i = 0; i < n; i++) {
            System.out.println("Nome do paciente:");
            String nome = scanner.nextLine();

            System.out.println("Data de nascimento (YYYY-MM-DD):");
            String dataNascimento = scanner.nextLine();

            System.out.println("Altura:");
            double altura = scanner.nextDouble();

            System.out.println("Peso:");
            double peso = scanner.nextDouble();

            System.out.println("Frequência Cardíaca:");
            double frequenciaCardiaca = scanner.nextDouble();

            System.out.println("Temperatura:");
            double temperatura = scanner.nextDouble();

            System.out.println("Saturação de Oxigénio:");
            double saturacaoOxigenio = scanner.nextDouble();
            scanner.nextLine();

            Paciente paciente = new Paciente(nome, dataNascimento, altura, peso, frequenciaCardiaca, temperatura, saturacaoOxigenio);
            gestor.adicionarPaciente(paciente);
        }

        System.out.println("Pacientes ordenados por data de nascimento:");
        gestor.listarPacientesOrdenadosPorDataNascimento();

        System.out.println("Deseja atualizar os sinais vitais de um paciente? (s/n)");
        String resposta = scanner.nextLine();

        if (resposta.equalsIgnoreCase("s")) {
            System.out.println("Digite o nome do paciente:");
            String nomePaciente = scanner.nextLine();

            for (Paciente p : gestor.getPacientes()) {  // Precisas de um getter para a lista de pacientes
                if (p.getNome().equalsIgnoreCase(nomePaciente)) {
                    System.out.println("Nova Frequência Cardíaca:");
                    double novaFrequencia = scanner.nextDouble();

                    System.out.println("Nova Temperatura:");
                    double novaTemperatura = scanner.nextDouble();

                    System.out.println("Nova Saturação de Oxigénio:");
                    double novaSaturacao = scanner.nextDouble();
                    scanner.nextLine();

                    p.registarSinaisVitais(novaFrequencia, novaTemperatura, novaSaturacao);
                    System.out.println("Sinais vitais atualizados! Nova classificação: " + p.classificarPaciente());
                    break;
                }
            }
        }
        FicheiroPacientes.guardarPacientes(gestor.getPacientes());

        System.out.println("Dados guardados no ficheiro.");

    }
}
